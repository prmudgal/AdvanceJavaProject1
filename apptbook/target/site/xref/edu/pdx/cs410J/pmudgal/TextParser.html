19  public class TextParser implements AppointmentBookParser {
20      String filename;
21      AppointmentBook appointmentBook;
22  
23      /**
24       * Parameterized Constructor
25       * @param filename : File containing appointments
26       * @param appointmentBook : ApointmentBook having appointments
27       */
28      public TextParser(String filename, AppointmentBook appointmentBook){
29          this.filename=filename;
30          this.appointmentBook=appointmentBook;
31      }
32  
33      /**
34       * This method is reading the contents of file and checking
35       * if everything in file is well formatted. And add the content
36       * of the file to appointmentBook.
37       * @return AbstractAppointmentBook
38       * @throws ParserException : Exception if issues in parsing
39       * the file.
40       */
41      @Override
42      public AbstractAppointmentBook parse() throws ParserException {
43          BufferedReader br =null;
44          try {
45              File file =new File(filename);
46              br=new BufferedReader(new FileReader(file));
47              for(String line : Files.readAllLines(Paths.get(filename))){
48                  Appointment appointment = new Appointment();
49                  String[] lineFromFile = line.split(",");
50                  if(line!=null && !line.isEmpty() && countDelimiters(line) ==3 && lineFromFile.length == 4) {
51                          appointment.setOwner(checkNullInsideFile(lineFromFile[0],"owner name"));
52                          appointment.setDescription(checkNullInsideFile(lineFromFile[1],"description"));
53                          appointment.setBeginTimeString(checkDateTimeFormatInFile(checkNullInsideFile(lineFromFile[2], "beginTime"),"beginTime"));
54                          appointment.setEndTimeString(checkDateTimeFormatInFile(checkNullInsideFile(lineFromFile[3],"endTime"), "endTime"));
55                          appointmentBook.setOwnerName(appointment.getOwner());
56                          appointmentBook.addAppointment(appointment);
57                  } else{
58                      throw new IOException("The file seems to be malformed." +
59                              " The file either has less/more commas or values in it." +
60                              " Please correct it and try again.");
61                  }
62              }
63          }  catch (ParseException|IOException e) {
64              System.out.println(e.getMessage());
65              System.exit(1);
66          } finally {
67              try {
68                  br.close();
69              } catch (IOException e) {
70                  System.out.println(e.getMessage());
71                  System.exit(1);
72              }
73          }
74          return appointmentBook;
75      }
76  
77      /**
78       * This method counts the number of deimiters which is comma.
79       * @param line : line from file
80       * @return counter: number of comma
81       */
82      private int countDelimiters(String line){
83          int counter = 0;
84          for (Character c: line.toCharArray()) {
85              if (c.equals(',')) {
86                  counter++;
87              }
88          }
89          return counter;
90      }
91  
92      /**
93       * This method checks if the passed string is not null
94       * and not empty and also make sure that it contains data.
95       * @param string : String to be checked as null
96       * @param fieldName : The fieldname from appointment
97       * @return the correct value or the error message
98       */
99      public static String checkNullInsideFile(String string, String fieldName){
100         if(string!=null && !string.trim().isEmpty() && !string.trim().equals("")){
101             return string;
102         }else {
103             System.out.println( fieldName + " is empty inside the file. Please correct the file and try again.");
104             System.exit(1);
105             return fieldName + " is empty inside the file. Please correct the file and try again.";
106 
107         }
108     }
109 
110     /**
111      * /**
112      * *This method checks for the correct format of date and time in the file.
113      *The date and time should be in format "mm/dd/yyyy hh:mm' or "m/d/yyyy hh:mm".
114      * This method also check for invalid dates e.g. 13/01/2015 or 12/40/2015.
115      * Also enforces the year has to pass with 4 digits.
116      * @param value : the date and time
117      * @param fieldName : beginTime or endTime fieldname
118      * @return : the correctly formatted date and time
119      * @throws ParseException : Exception while parsing the date and time
120      */
121 
122     public static String checkDateTimeFormatInFile(String value, String fieldName) throws ParseException {
123         if (value == null || !value.matches("^\\d{1,2}/\\d{1,2}/\\d{4} \\d{1,2}:\\d{2}$")) {
124             System.out.println("The date and time for "+ fieldName+" in file is not in format mm/dd/yyyy hh:mm. Please correct it and try again");
125             System.exit(1);
126         }else{
127             SimpleDateFormat df = new SimpleDateFormat("MM/dd/yyyy HH:mm");
128             df.setLenient(false);
129             df.parse(value);
130         }
131         return value;
132     }
133 }
