001package edu.pdx.cs410J.pmudgal;
002
003import edu.pdx.cs410J.AbstractAppointmentBook;
004import edu.pdx.cs410J.AppointmentBookDumper;
005
006import java.io.*;
007import java.util.ArrayList;
008
009/**
010 * Created by Priyanka on 7/6/2016.
011 */
012public class TextDumper implements AppointmentBookDumper {
013    String filename;
014
015    /**
016     * Parameterized constructor
017     * @param filename : File in which appointments are to to be written.
018     */
019    public TextDumper(String filename){
020        this.filename=filename;
021    }
022
023    /**
024     * Overridden method of AppointmentBookDumper.
025     * This method writes the appointments to the
026     * file by calling writeToFile() method.
027     * @param abstractAppointmentBook : instance of AppointmentBook
028     * @throws IOException : Handles IO Exception in case of file wrting.
029     */
030    @Override
031    public void dump(AbstractAppointmentBook abstractAppointmentBook) throws IOException {
032        try {
033            //Whatever the file path is.
034            File file = new File(filename);
035                System.out.println("Created new file");
036                FileOutputStream is = new FileOutputStream(file);
037                OutputStreamWriter osw = new OutputStreamWriter(is);
038                BufferedWriter w = new BufferedWriter(osw);
039                for(Appointment appointment:(new ArrayList<Appointment>(abstractAppointmentBook.getAppointments()))){
040                    writeToFile(w, appointment);
041                }
042                w.close();
043        } catch (IOException e) {
044            e.getMessage();
045            System.err.println("Problem writing to the file :" + filename);
046            e.printStackTrace();
047        }
048    }
049
050    /**
051     * This method takes the individual field from appointment
052     * and writes it to file using bufferedWriter separated by comma.
053     * @param w : BfferedWriter
054     * @param appointment : instance of Appointment
055     * @throws IOException : handles while writing file.
056     */
057    private static void writeToFile(BufferedWriter w, Appointment appointment) throws IOException {
058        w.write(appointment.getOwner());
059        w.write(",");
060        w.write(appointment.getDescription());
061        w.write(",");
062        w.write(appointment.getBeginTimeString());
063        w.write(",");
064        w.write(appointment.getEndTimeString());
065        w.newLine();
066    }
067
068}
