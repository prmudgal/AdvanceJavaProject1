039    /**
040     *This method takes the arguments passed form command line. It checks if the input string is options or arguments.
041     * The passed arguments for sppointments are added to appointmentBook.
042     * It also checks if the beginTime and endTime are in correct format.
043     * In any error scenario case, and error message is shown and program exits.
044     * @param args : Command line Arguments
045     * @param appointmentBook : the appointment book which would contain appointments
046     * @return : the correctly formatted appointment with all valid values for the attributes.
047     * @throws ParseException : In case of any exception while parsing the dates.
048     */
049    public static AppointmentBook prepareAppointmentBook(String[] args, AppointmentBook appointmentBook) throws Exception {
050        Appointment appointment = new Appointment();
051
052        if(Arrays.asList(args).contains("-README")){
053            displayReadme();
054        } else {
055            for (int i = 0; i < args.length; i++) {
056                if (args[i].startsWith("-")) {
057                    if (args[i].contains(Constants.PRINT)) {
058                        System.out.println("Printing the Appointment : " + (args.length > 2 ? args[2] : "No description provided in command line."));
059                        i++;
060                    } else {
061                        System.out.println(args[i] + ": This is not a correct option. Please provide the correct option.");
062                        System.exit(1);
063                    }
064                }
065                if (args.length - i == 8) {
066                    appointment.setOwner(validateOwnerName(args[i])); //Checks if the argument starts with "-", it is not considered as owner name
067                    appointment.setDescription(checkNull(args[++i], "description")); //
068                    appointment.setBeginTimeString(checkDateTimeFormatWithAmPm(checkNull(args[++i], "beginDateTime").concat(" ").concat(checkNull(args[++i], "beginDateTime")).concat(" ").concat(checkNull(args[++i], "beginDateTime"))));
069                    appointment.setEndTimeString(checkDateTimeFormatWithAmPm(checkNull(args[++i], "endDateTime").concat(" ").concat(checkNull(args[++i], "endDateTime")).concat(" ").concat(checkNull(args[++i], "endDateTime"))));
070                    if (appointmentBook.getOwnerName() != null && !appointmentBook.getOwnerName().isEmpty() &&
071                            !appointmentBook.getOwnerName().equals(appointment.getOwner())) {
072                        throw new Exception("Owner name passed in argument " + " '" + appointment.getOwner() + "' " + " is different than the one present in AppointmentBook " + " '" + appointmentBook.getOwnerName() + "' ");
073                    }
074                    appointmentBook.addAppointment(appointment);
075                    System.out.println(appointment.toString());
076                } else if (args.length - i < 8) {
077                    System.out.println("Some arguments are missing. Please provide complete arguments : owner description beginDateTime endDateTime");
078                    System.exit(1);
079                } else if (args.length - i > 8) {
080                    System.out.println("There are extra arguments provided than required arguments.\n" +
081                            Constants.PROVIDE_ARG);
082                    System.exit(1);
083                }
084            }
085        }
086        return appointmentBook;
087    }

131    /**
132     * *This method checks for the correct format of date and time.
133     *The date and time should be in format "mm/dd/yyyy hh:mm' or "m/d/yyyy hh:mm".
134     * This method also check for invalid dates e.g. 13/01/2015 or 12/40/2015.
135     * Also enforces the year has to pass with 4 digits.
136     * @param value : the date and time
137     * @return : the correctly formatted date and time
138     * @throws ParseException : Exception while parsing the date and time
139     */
140    public static String checkDateTimeFormat(String value) throws ParseException {
141        if (value == null || !value.matches("^\\d{1,2}/\\d{1,2}/\\d{4} \\d{1,2}:\\d{2}$")) {
142            System.out.println("Please provide the date and time in format mm/dd/yyyy hh:mm");
143            System.exit(1);
144        }else{
145            SimpleDateFormat df = new SimpleDateFormat("MM/dd/yyyy HH:mm");
146            df.setLenient(false);
147            df.parse(value);
148        }
149        return value;
150    }

170    /**
171     * *This method checks for the correct format of date and time.
172     *The date and time should be in format "mm/dd/yyyy hh:mm am/pm' or "m/d/yyyy hh:mm am/pm".
173     * This method also check for invalid dates e.g. 13/01/2015 or 12/40/2015.
174     * Also enforces the year has to pass with 4 digits.
175     * @param value : the date and time
176     * @return : the correctly formatted date and time
177     * @throws ParseException : Exception while parsing the date and time
178     */
179    public static String checkDateTimeFormatWithAmPm(String value) throws ParseException {
180        if (value == null || !value.matches("^\\d{1,2}/\\d{1,2}/\\d{4} \\d{1,2}:\\d{2} [aApP][mM]$")) {
181            System.out.println("Please provide the date and time in format mm/dd/yyyy hh:mm");
182            System.exit(1);
183        }else{
184            SimpleDateFormat df = new SimpleDateFormat("MM/dd/yyyy hh:mm a");
185            df.setLenient(false);
186            df.parse(value);
187        }
188        return value;
189    }
