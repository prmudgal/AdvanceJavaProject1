106     private static void prepareAppointmentBook(String[] args) throws Exception {
107         try {
108             ArrayList<String> argList = new ArrayList<String>(Arrays.asList(args));
109             String host = null;
110             Integer port = null;

113             String beginTime = null;
114             String endTime = null;
115             boolean searchFlag = false;
116             if (args.length != 0) {
117                 if (argList.contains(Constants.README)) {
118                     displayReadMe();

120                     if (!argList.contains(Constants.HOST)) {
121                         usage("Missing Host");
122 
123                     } else if (!argList.contains(Constants.PORT)) {
124                         usage("Missing port");
125                     }
126                     int hostIndex = argList.indexOf(Constants.HOST);
127                     host = argList.get(hostIndex + 1);
128                     argList.remove(hostIndex + 1);
129                     argList.remove(hostIndex);
130                     if(host.equals(Constants.PORT)){
131                         usage("Please provide the host. ");
132                     }
133                     int portIndex = argList.indexOf(Constants.PORT);
134                     String portString = argList.get(portIndex + 1);
135                     argList.remove(portIndex + 1);
136                     argList.remove(portIndex);

145                         return;
146                     }
147                     AppointmentBookRestClient client = new AppointmentBookRestClient(host, port);
148                     String[] modifiedArgs = argList.toArray(new String[argList.size()]);
149                     for (int i = 0; i < modifiedArgs.length; i++) {

256     /**
257      * *This method checks for the correct format of date and time.
258      *The date and time should be in format "mm/dd/yyyy hh:mm am/pm' or "m/d/yyyy hh:mm am/pm".
259      * This method also check for invalid dates e.g. 13/01/2015 or 12/40/2015.
260      * Also enforces the year has to pass with 4 digits.
261      * @param value : the date and time
262      * @param fieldName : fieldName
263      * @return : the correctly formatted date and time
264      * @throws ParseException : Exception while parsing the date and time
265      */
266     public static String checkDateTimeFormatWithAmPm(String value, String fieldName) throws ParseException {
267         if (value == null || !value.matches("^\\d{1,2}/\\d{1,2}/\\d{4} \\d{1,2}:\\d{2} [aApP][mM]$")) {
268             System.out.println(Messages.noFormattedDate(fieldName));
269             System.exit(1);
270         }else{
271             SimpleDateFormat df = new SimpleDateFormat("MM/dd/yyyy hh:mm a");
272             df.setLenient(false);
273             df.parse(value);
274         }
275         return value;
276     }
