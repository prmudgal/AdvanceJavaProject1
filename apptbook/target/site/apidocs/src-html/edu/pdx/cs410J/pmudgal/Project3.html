132    /**
133     * *This method checks for the correct format of date and time.
134     *The date and time should be in format "mm/dd/yyyy hh:mm am/pm' or "m/d/yyyy hh:mm am/pm".
135     * This method also check for invalid dates e.g. 13/01/2015 or 12/40/2015.
136     * Also enforces the year has to pass with 4 digits.
137     * @param value : the date and time
138     * @return : the correctly formatted date and time
139     * @throws ParseException : Exception while parsing the date and time
140     */
141    public static String checkDateTimeFormat(String value) throws ParseException {
142        if (value == null || !value.matches("^\\d{1,2}/\\d{1,2}/\\d{4} \\d{1,2}:\\d{2} [aApP][mM]$")) {
143            System.out.println("Please provide the date and time in format mm/dd/yyyy hh:mm");
144            System.exit(1);
145        }else{
146            SimpleDateFormat df = new SimpleDateFormat("MM/dd/yyyy hh:mm a");
147            df.setLenient(false);
148            df.parse(value);
149        }
150        return value;
151    }
