021        parseCommandLineArgs(args);
022
023    }
024
025    /**
026     * This method checks if there is "-textfile" option present in
027     * command line arguments. If yes, then it checks if the filename
028     * passed in command line args, exists. If no, it created the file.
029     * If yes, it writes the content of appointment in that file by calling
030     * dump() method of TextDumper.
031     * If command line arguments does not have "-textfile" option,
032     * the project processes same as Project 1.
033     * @param args : Command line arguments passed.
034     */
035    private static void parseCommandLineArgs(String[] args) {
036        Project1 project1=new Project1();
037        AppointmentBook appointmentBook = new AppointmentBook();
038
039        try {
040            ArrayList<String> arrayList= new ArrayList<String>(Arrays.asList(args));
041            if(!arrayList.contains(Constants.README) ){
042                if (arrayList.contains(Constants.TEXTFILE)) {
043                    int index = arrayList.indexOf(Constants.TEXTFILE);
044                    String filename = arrayList.get(index + 1);
045                    arrayList.remove(index + 1);
046                    arrayList.remove(Constants.TEXTFILE);
047
048                    String[] stockArr = new String[arrayList.size()];
049                    stockArr = arrayList.toArray(new String[arrayList.size()]);
050                    File file = new File(filename);
051                    if (file.createNewFile()) {
052                        appointmentBook = parseAndPrepareTheContentsOfFile(filename, stockArr);
053                        TextDumper textDumper = new TextDumper(filename);
054                        textDumper.dump(appointmentBook);
055                        System.out.println("Created new file : "+ filename);
056                    } else {
057                        appointmentBook = parseAndPrepareTheContentsOfFile(filename, stockArr);
058                        TextDumper textDumper = new TextDumper(filename);
059                        textDumper.dump(appointmentBook);
060                        System.out.println("Written appointment to file : "+ filename);
061                    }
062                } else {
063                    project1.prepareAppointmentBook(args, appointmentBook);
064                }
065            }else{
066                System.out.println(Constants.PROJECT2_README_DESC);
067            }
068        } catch (ParseException e) {
069            System.out.println(e.getMessage());
070            System.exit(1);
071        } catch (IOException e) {
072            System.out.println(e.getMessage());
073            System.exit(1);
074        } catch (ParserException e) {
075            System.out.println(e.getMessage());
076            System.exit(1);
077        } catch (Exception e) {
078            System.out.println(e.getMessage());
079            System.exit(1);
080        }
081    }
082
083    /**
084     * This method calls the parse() method of TextParser by passing
085     * filename and command line args. After parsing, it also calls
086     * prepareAppointmentBook() method of Project1.
087     * @param filename : Filename which contains appointments
088     * @param args : command line args
089     * @return : AppointmentBook
090     * @throws Exception : In case of any exception, regarding parsing the file.
091     */
092    public static AppointmentBook parseAndPrepareTheContentsOfFile(String filename, String[] args) throws Exception {
093        Project1 project1 =new Project1();
094        AbstractAppointmentBook appointmentBook = new AppointmentBook();
095        TextParser textParser = new TextParser(filename, (AppointmentBook) appointmentBook);
096        appointmentBook = textParser.parse();
097        appointmentBook = project1.prepareAppointmentBook(args, (AppointmentBook) appointmentBook);
098        return (AppointmentBook) appointmentBook;
099    }
