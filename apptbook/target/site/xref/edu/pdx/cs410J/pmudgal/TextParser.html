100     /**
101      * This method counts the number of deimiters which is comma.
102      * @param line : line from file
103      * @return counter: number of comma
104      */
105     private int countDelimiters(String line){
106         int counter = 0;
107         for (Character c: line.toCharArray()) {
108             if (c.equals(',')) {
109                 counter++;
110             }
111         }
112         return counter;
113     }

115     /**
116      * This method checks if the passed string is not null
117      * and not empty and also make sure that it contains data.
118      * @param string : String to be checked as null
119      * @param fieldName : The fieldname from appointment
120      * @return the correct value or the error message
121      */
122     public static String checkNullInsideFile(String string, String fieldName){
123         if(string!=null && !string.trim().isEmpty() && !string.trim().equals("")){
124             return string;
125         }else {
126             System.out.println( fieldName + " is empty inside the file. Please correct the file and try again.");

128             return fieldName + " is empty inside the file. Please correct the file and try again.";

130         }

133     /**
134      * /**
135      * *This method checks for the correct format of date and time in the file.
136      *The date and time should be in format "mm/dd/yyyy hh:mm' or "m/d/yyyy hh:mm".
137      * This method also check for invalid dates e.g. 13/01/2015 or 12/40/2015.
138      * Also enforces the year has to pass with 4 digits.
139      * @param value : the date and time
140      * @param fieldName : beginTime or endTime fieldname
141      * @return : the correctly formatted date and time
142      * @throws ParseException : Exception while parsing the date and time
143      */
144 
145     public static String checkDateTimeFormatInFile(String value, String fieldName) throws ParseException {
146         if (value == null || !value.matches("^\\d{1,2}/\\d{1,2}/\\d{4} \\d{1,2}:\\d{2}$")) {
147             System.out.println("The date and time for "+ fieldName+" in file is not in format mm/dd/yyyy hh:mm. Please correct it and try again");
148             System.exit(1);
149         }else{
150             SimpleDateFormat df = new SimpleDateFormat("MM/dd/yyyy HH:mm");
151             df.setLenient(false);
152             df.parse(value);
153         }
154         return value;
155     }

158     /**
159      * *This method checks for the correct format of date and time in the file.
160      *The date and time should be in format "mm/dd/yyyy hh:mm am/pm' or "m/d/yyyy hh:mm am/pm".
161      * This method also check for invalid dates e.g. 13/01/2015 or 12/40/2015.
162      * Also enforces the year has to pass with 4 digits.
163      * @param value : the date and time
164      * @param fieldName: the field name such as owner name/desciption.
165      * @return : the correctly formatted date and time
166      * @throws ParseException : Exception while parsing the date and time
167      */
168     public static String checkDateTimeFormatWithAmPmInFile(String value, String fieldName) throws ParseException {
169         if (value == null || !value.matches("^\\d{1,2}/\\d{1,2}/\\d{2} \\d{1,2}:\\d{2} [aApP][mM]$")) {
170             System.out.println("The date and time for "+ fieldName+" in file is not in format mm/dd/yyyy hh:mm am/pm. Please correct it and try again");
171             System.exit(1);
172         }else{
173             SimpleDateFormat df = new SimpleDateFormat("MM/dd/yy hh:mm a");
174             df.setLenient(false);
175             df.parse(value);
176         }
177         return value;
178     }
