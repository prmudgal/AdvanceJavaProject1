003import java.io.PrintWriter;
004import java.text.ParseException;
005import java.text.SimpleDateFormat;
006
007/**
008 * Created by Priyanka on 7/23/2016.
009 */
010public class DateValidator {
011    /**
012     * *This method checks for the correct format of date and time.
013     *The date and time should be in format "mm/dd/yyyy hh:mm am/pm' or "m/d/yyyy hh:mm am/pm".
014     * This method also check for invalid dates e.g. 13/01/2015 or 12/40/2015.
015     * Also enforces the year has to pass with 4 digits.
016     * @param value : the date and time
017     * @param pw : Print Writer
018     * @return : the correctly formatted date and time
019     * @throws ParseException : Exception while parsing the date and time
020     */
021    public static String checkDateTimeFormat(String value, PrintWriter pw) throws ParseException {
022        if (value == null || !value.matches("^\\d{1,2}/\\d{1,2}/\\d{4} \\d{1,2}:\\d{2} [aApP][mM]$")) {
023//            pw.println("Please provide the date and time in format mm/dd/yyyy hh:mm");
024            return "false";
025        }else{
026            SimpleDateFormat df = new SimpleDateFormat("MM/dd/yyyy hh:mm a");
027            df.setLenient(false);
028            df.parse(value);
029        }
030        return value;
031    }
032}
