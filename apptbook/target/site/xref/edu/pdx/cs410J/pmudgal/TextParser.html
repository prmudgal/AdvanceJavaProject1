19  public class TextParser implements AppointmentBookParser {
20      String filename;
21      AppointmentBook appointmentBook;
22  
23      /**
24       * Parameterized Constructor
25       * @param filename : File containing appointments
26       * @param appointmentBook : ApointmentBook having appointments
27       */
28      public TextParser(String filename, AppointmentBook appointmentBook){
29          this.filename=filename;
30          this.appointmentBook=appointmentBook;
31      }
32  
33      /**
34       * This method is reading the contents of file and checking
35       * if everything in file is well formatted. And add the content
36       * of the file to appointmentBook.
37       * @return AbstractAppointmentBook
38       * @throws ParserException : Exception if issues in parsing
39       * the file.
40       */
41      @Override
42      public AbstractAppointmentBook parse() throws ParserException {
43          BufferedReader br =null;
44          try {
45              File file =new File(filename);
46              if(file.exists() && file.length() == 0){
47                  throw new IOException("The file seems to be exist, " +
48                          "but it is empty and considered as malformatted. We ca not write into this file. ");
49              }
50             br=new BufferedReader(new FileReader(file));
51              for(String line : Files.readAllLines(Paths.get(filename))){
52                  System.out.println(" inside for : " + line);
53                  StringTokenizer stringTokenizer = new StringTokenizer(line, ",");
54                  if(line!=null && !line.isEmpty() && countDelimiters(line) ==3) {
55                      Appointment appointment = new Appointment();
56                      while (stringTokenizer.hasMoreTokens()) {
57                          appointment.setOwner(Project1.checkNull(stringTokenizer.nextToken(),"owner name"));
58                          appointment.setDescription(Project1.checkNull(stringTokenizer.nextToken(),"description"));
59                          appointment.setBeginTimeString(Project1.checkDateTimeFormat(stringTokenizer.nextToken()));
60                          appointment.setEndTimeString(Project1.checkDateTimeFormat(stringTokenizer.nextToken()));
61                          appointmentBook.setOwnerName(appointment.getOwner());
62                          appointmentBook.addAppointment(appointment);
63                      }
64                  } else{
65                      throw new IOException("The file seems to be malformed." +
66                              " The file either has more commas or values in it." +
67                              " Please correct it and try again.");
68  
69                  }
70              }
71          }  catch (ParseException|IOException e) {
72              System.out.println(e.getMessage());
73              System.exit(1);
74          } finally {
75              try {
76                  br.close();
77              } catch (IOException e) {
78                  System.out.println(e.getMessage());
79                  System.exit(1);
80              }
81          }
82          System.out.println(appointmentBook.getAppointments().size() + " Size");
83          return appointmentBook;
84      }
85  
86      /**
87       * This method counts the number of deimiters which is comma.
88       * @param line : line from file
89       * @return counter: number of comma
90       */
91      private int countDelimiters(String line){
92          int counter = 0;
93          for (Character c: line.toCharArray()) {
94              if (c.equals(',')) {
95                  counter++;
96              }
97          }
98          return counter;
99      }
100 }
