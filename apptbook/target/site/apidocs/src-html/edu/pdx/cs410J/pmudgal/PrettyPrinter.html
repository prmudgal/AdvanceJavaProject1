003import edu.pdx.cs410J.AbstractAppointmentBook;
004import edu.pdx.cs410J.AppointmentBookDumper;
005
006import java.io.IOException;
007import java.io.PrintWriter;
008
009/**
010 * Created by Priyanka on 7/21/2016.
011 */
012public class PrettyPrinter implements AppointmentBookDumper {
013
014    private final PrintWriter writer;
015
016    /**
017     * Parameterized constructor
018     * @param writer : Print Writer
019     */
020    public PrettyPrinter(PrintWriter writer){
021        this.writer=writer;
022    }
023
024    /**
025     * Write the appointments
026     * @param abstractAppointmentBook : AppointmentBook
027     * @throws IOException : Any Exception while writing
028     */
029    @Override
030    public void dump(AbstractAppointmentBook abstractAppointmentBook) throws IOException {
031        if (abstractAppointmentBook != null && abstractAppointmentBook.getAppointments().size()!=0) {
032            writer.println(abstractAppointmentBook.getAppointments());
033//            System.out.println(abstractAppointmentBook.getAppointments());
034        }else{
035            writer.println(Messages.getMappingCount(0));
036//            System.out.println(Messages.getMappingCount(0));
037        }
038    }
039}
