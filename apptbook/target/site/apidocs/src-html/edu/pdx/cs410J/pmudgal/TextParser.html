001package edu.pdx.cs410J.pmudgal;
002
003import edu.pdx.cs410J.AbstractAppointmentBook;
004import edu.pdx.cs410J.AppointmentBookParser;
005import edu.pdx.cs410J.ParserException;
006
007import java.io.BufferedReader;
008import java.io.File;
009import java.io.FileReader;
010import java.io.IOException;
011import java.nio.file.Files;
012import java.nio.file.Paths;
013import java.text.ParseException;
014import java.util.StringTokenizer;
015
016/**
017 * Created by Priyanka on 7/6/2016.
018 */
019public class TextParser implements AppointmentBookParser {
020    String filename;
021    AppointmentBook appointmentBook;
022
023    /**
024     * Parameterized Constructor
025     * @param filename : File containing appointments
026     * @param appointmentBook : ApointmentBook having appointments
027     */
028    public TextParser(String filename, AppointmentBook appointmentBook){
029        this.filename=filename;
030        this.appointmentBook=appointmentBook;
031    }
032
033    /**
034     * This method is reading the contents of file and checking
035     * if everything in file is well formatted. And add the content
036     * of the file to appointmentBook.
037     * @return AbstractAppointmentBook
038     * @throws ParserException : Exception if issues in parsing
039     * the file.
040     */
041    @Override
042    public AbstractAppointmentBook parse() throws ParserException {
043        BufferedReader br =null;
044        try {
045            File file =new File(filename);
046            if(file.exists() && file.length() == 0){
047                throw new IOException("The file seems to be exist, " +
048                        "but it is empty and considered as malformatted. We ca not write into this file. ");
049            }
050           br=new BufferedReader(new FileReader(file));
051            for(String line : Files.readAllLines(Paths.get(filename))){
052                System.out.println(" inside for : " + line);
053                StringTokenizer stringTokenizer = new StringTokenizer(line, ",");
054                if(line!=null && !line.isEmpty() && countDelimiters(line) ==3) {
055                    Appointment appointment = new Appointment();
056                    while (stringTokenizer.hasMoreTokens()) {
057                        appointment.setOwner(Project1.checkNull(stringTokenizer.nextToken(),"owner name"));
058                        appointment.setDescription(Project1.checkNull(stringTokenizer.nextToken(),"description"));
059                        appointment.setBeginTimeString(Project1.checkDateTimeFormat(stringTokenizer.nextToken()));
060                        appointment.setEndTimeString(Project1.checkDateTimeFormat(stringTokenizer.nextToken()));
061                        appointmentBook.setOwnerName(appointment.getOwner());
062                        appointmentBook.addAppointment(appointment);
063                    }
064                } else{
065                    throw new IOException("The file seems to be malformed." +
066                            " The file either has more commas or values in it." +
067                            " Please correct it and try again.");
068
069                }
070            }
071        }  catch (ParseException|IOException e) {
072            System.out.println(e.getMessage());
073            System.exit(1);
074        } finally {
075            try {
076                br.close();
077            } catch (IOException e) {
078                System.out.println(e.getMessage());
079                System.exit(1);
080            }
081        }
082        System.out.println(appointmentBook.getAppointments().size() + " Size");
083        return appointmentBook;
084    }
085
086    /**
087     * This method counts the number of deimiters which is comma.
088     * @param line : line from file
089     * @return counter: number of comma
090     */
091    private int countDelimiters(String line){
092        int counter = 0;
093        for (Character c: line.toCharArray()) {
094            if (c.equals(',')) {
095                counter++;
096            }
097        }
098        return counter;
099    }
100}
