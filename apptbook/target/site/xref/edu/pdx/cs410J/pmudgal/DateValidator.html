10  public class DateValidator {
11      /**
12       * *This method checks for the correct format of date and time.
13       *The date and time should be in format "mm/dd/yyyy hh:mm am/pm' or "m/d/yyyy hh:mm am/pm".
14       * This method also check for invalid dates e.g. 13/01/2015 or 12/40/2015.
15       * Also enforces the year has to pass with 4 digits.
16       * @param value : the date and time
17       * @param pw : Print Writer
18       * @return : the correctly formatted date and time
19       * @throws ParseException : Exception while parsing the date and time
20       */
21      public static String checkDateTimeFormat(String value, PrintWriter pw) throws ParseException {
22          if (value == null || !value.matches("^\\d{1,2}/\\d{1,2}/\\d{4} \\d{1,2}:\\d{2} [aApP][mM]$")) {
23  //            pw.println("Please provide the date and time in format mm/dd/yyyy hh:mm");
24              return "false";
25          }else{
26              SimpleDateFormat df = new SimpleDateFormat("MM/dd/yyyy hh:mm a");
27              df.setLenient(false);
28              df.parse(value);
29          }
30          return value;
31      }
32  }
