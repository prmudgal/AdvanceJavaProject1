001package edu.pdx.cs410J.pmudgal;
002
003import edu.pdx.cs410J.AbstractAppointmentBook;
004import edu.pdx.cs410J.AppointmentBookParser;
005import edu.pdx.cs410J.ParserException;
006
007import java.io.BufferedReader;
008import java.io.File;
009import java.io.FileReader;
010import java.io.IOException;
011import java.nio.file.Files;
012import java.nio.file.Paths;
013import java.text.ParseException;
014import java.text.SimpleDateFormat;
015
016/**
017 * Created by Priyanka on 7/6/2016.
018 */
019public class TextParser implements AppointmentBookParser {
020    String filename;
021    AppointmentBook appointmentBook;
022
023    /**
024     * Parameterized Constructor
025     * @param filename : File containing appointments
026     * @param appointmentBook : ApointmentBook having appointments
027     */
028    public TextParser(String filename, AppointmentBook appointmentBook){
029        this.filename=filename;
030        this.appointmentBook=appointmentBook;
031    }
032
033    /**
034     * This method is reading the contents of file and checking
035     * if everything in file is well formatted. And add the content
036     * of the file to appointmentBook.
037     * @return AbstractAppointmentBook
038     * @throws ParserException : Exception if issues in parsing
039     * the file.
040     */
041    @Override
042    public AbstractAppointmentBook parse() throws ParserException {
043        BufferedReader br =null;
044        try {
045            File file =new File(filename);
046            br=new BufferedReader(new FileReader(file));
047            for(String line : Files.readAllLines(Paths.get(filename))){
048                Appointment appointment = new Appointment();
049                String[] lineFromFile = line.split(",");
050                if(line!=null && !line.isEmpty() && countDelimiters(line) ==3 && lineFromFile.length == 4) {
051                        appointment.setOwner(checkNullInsideFile(lineFromFile[0],"owner name"));
052                        appointment.setDescription(checkNullInsideFile(lineFromFile[1],"description"));
053                        appointment.setBeginTimeString(checkDateTimeFormatInFile(checkNullInsideFile(lineFromFile[2], "beginTime"),"beginTime"));
054                        appointment.setEndTimeString(checkDateTimeFormatInFile(checkNullInsideFile(lineFromFile[3],"endTime"), "endTime"));
055                        appointmentBook.setOwnerName(appointment.getOwner());
056                        appointmentBook.addAppointment(appointment);
057                } else{
058                    throw new IOException("The file seems to be malformed." +
059                            " The file either has less/more commas or values in it." +
060                            " Please correct it and try again.");
061                }
062            }
063        }  catch (ParseException|IOException e) {
064            System.out.println(e.getMessage());
065            System.exit(1);
066        } finally {
067            try {
068                br.close();
069            } catch (IOException e) {
070                System.out.println(e.getMessage());
071                System.exit(1);
072            }
073        }
074        return appointmentBook;
075    }
076
077    /**
078     * This method counts the number of deimiters which is comma.
079     * @param line : line from file
080     * @return counter: number of comma
081     */
082    private int countDelimiters(String line){
083        int counter = 0;
084        for (Character c: line.toCharArray()) {
085            if (c.equals(',')) {
086                counter++;
087            }
088        }
089        return counter;
090    }
091
092    /**
093     * This method checks if the passed string is not null
094     * and not empty and also make sure that it contains data.
095     * @param string : String to be checked as null
096     * @param fieldName : The fieldname from appointment
097     * @return the correct value or the error message
098     */
099    public static String checkNullInsideFile(String string, String fieldName){
100        if(string!=null && !string.trim().isEmpty() && !string.trim().equals("")){
101            return string;
102        }else {
103            System.out.println( fieldName + " is empty inside the file. Please correct the file and try again.");
104            System.exit(1);
105            return fieldName + " is empty inside the file. Please correct the file and try again.";
106
107        }
108    }
109
110    /**
111     * /**
112     * *This method checks for the correct format of date and time in the file.
113     *The date and time should be in format "mm/dd/yyyy hh:mm' or "m/d/yyyy hh:mm".
114     * This method also check for invalid dates e.g. 13/01/2015 or 12/40/2015.
115     * Also enforces the year has to pass with 4 digits.
116     * @param value : the date and time
117     * @param fieldName : beginTime or endTime fieldname
118     * @return : the correctly formatted date and time
119     * @throws ParseException : Exception while parsing the date and time
120     */
121
122    public static String checkDateTimeFormatInFile(String value, String fieldName) throws ParseException {
123        if (value == null || !value.matches("^\\d{1,2}/\\d{1,2}/\\d{4} \\d{1,2}:\\d{2}$")) {
124            System.out.println("The date and time for "+ fieldName+" in file is not in format mm/dd/yyyy hh:mm. Please correct it and try again");
125            System.exit(1);
126        }else{
127            SimpleDateFormat df = new SimpleDateFormat("MM/dd/yyyy HH:mm");
128            df.setLenient(false);
129            df.parse(value);
130        }
131        return value;
132    }
133}
