001package edu.pdx.cs410J.pmudgal;
002
003/**
004 * Class for formatting messages on the server side.  This is mainly to enable
005 * test methods that validate that the server returned expected strings.
006 */
007public class Messages
008{
009    /**
010     * Mapping Count
011     * @param count : Int
012     * @return Message
013     */
014    public static String getMappingCount( int count )
015    {
016        return String.format( "Server contains %d Appointments", count );
017    }
018
019    /**
020     * Format the key value
021     * @param key : Key
022     * @param value : Value
023     * @return : Message
024     */
025    public static String formatKeyValuePair( String key, String value )
026    {
027        return String.format("  %s -> %s", key, value);
028    }
029
030    /**
031     * Missing required Paramateres
032     * @param parameterName : parameter Name
033     * @return : Message
034     */
035    public static String missingRequiredParameter( String parameterName )
036    {
037        return String.format("The required parameter \"%s\" is missing", parameterName);
038    }
039
040    /**
041     * Date not formatted
042     * @param parameterName : parameter name
043     * @return : Message
044     */
045    public static String noFormattedDate( String parameterName )
046    {
047        return String.format("The dateTime parameter \"%s\" is not in format mm/dd/yyyy hh:mm a", parameterName);
048    }
049
050    /**
051     * Mapped key value
052     * @param key : Key
053     * @param value : Value
054     * @return Message
055     */
056    public static String mappedKeyValue( String key, String value )
057    {
058        return String.format( "Mapped %s to %s", key, value );
059    }
060
061    /**
062     * Mapped owner to Appointment Book
063     * @param owner : Owner name
064     * @param appointmentBook : app book
065     * @return Message
066     */
067    public static String mappedOwnerToAppointmentBook( String owner, String appointmentBook )
068    {
069        return String.format( "Added appointment :  %s has %s", owner, appointmentBook );
070    }
071
072    /**
073     * Delete all mapping
074     * @return : Message
075     */
076    public static String allMappingsDeleted() {
077        return "All mappings have been deleted";
078    }
079}
