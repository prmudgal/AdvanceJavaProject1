38      /**
39       *This method takes the arguments passed form command line. It checks if the input string is options or arguments.
40       * The passed arguments for sppointments are added to appointmentBook.
41       * It also checks if the beginTime and endTime are in correct format.
42       * In any error scenario case, and error message is shown and program exits.
43       * @param args : Command line Arguments
44       * @param appointmentBook : the appointment book which would contain appointments
45       * @return : the correctly formatted appointment with all valid values for the attributes.
46       * @throws ParseException : In case of any exception while parsing the dates.
47       */
48      public static AppointmentBook prepareAppointmentBook(String[] args, AppointmentBook appointmentBook) throws Exception {
49          Appointment appointment = new Appointment();
50  
51          if(args.length==0){
52              System.out.println("You did not provide any arguments. Please provide the missing arguments in format: "+Constants.usage);
53          }else if(Arrays.asList(args).contains("-README")){
54              displayReadme();
55          } else {
56              for (int i = 0; i < args.length; i++) {
57                  if (args[i].startsWith("-")) {
58                      if (args[i].contains(Constants.PRINT)) {
59                          System.out.println("Printing the Appointment : " + (args.length>2?args[2]:"No description provided in command line."));
60                          i++;
61                      } else {
62                          System.out.println(args[i]+ ": This is not a correct option. Please provide the correct option.");
63                          System.exit(1);
64                      }
65                  }
66                  if ( args.length - i == 6) {
67                      appointment.setOwner(validateOwnerName(args[i])); //Checks if the argument starts with "-", it is not considered as owner name
68                      appointment.setDescription(checkNull(args[++i], "description")); //
69                      appointment.setBeginTimeString(checkDateTimeFormat(checkNull(args[++i], "beginDateTime").concat(" ").concat(checkNull(args[++i], "beginDateTime"))));
70                      appointment.setEndTimeString(checkDateTimeFormat(checkNull(args[++i], "endDateTime").concat(" ").concat(checkNull(args[++i], "endDateTime"))));
71                      if(appointmentBook.getOwnerName()!=null && !appointmentBook.getOwnerName().isEmpty() &&
72                              !appointmentBook.getOwnerName().equals(appointment.getOwner())){
73                          throw new Exception("Owner name is different than the one in Book");
74                      }
75                      appointmentBook.addAppointment(appointment);
76                      System.out.println(appointment.getOwner());
77                      System.out.println(appointment.getDescription());
78                      System.out.println(appointment.getBeginTimeString());
79                      System.out.println(appointment.getEndTimeString());
80                      System.out.println(appointment.toString());
81                  } else if( args.length - i < 6){
82                      System.out.println("Some arguments are missing. Please provide complete arguments : owner description beginDateTime endDateTime");
83                      System.exit(1);
84                  }else if( args.length - i > 6){
85                      System.out.println("There are extra arguments provided than required arguments.\n" +
86                              Constants.PROVIDE_ARG);
87                      System.exit(1);
88                  }
89              }
90  
91          }
92          return appointmentBook;
93      }

137     /**
138      * *This method checks for the correct format of date and time.
139      *The date and time should be in format "mm/dd/yyyy hh:mm' or "m/d/yyyy hh:mm".
140      * This method also check for invalid dates e.g. 13/01/2015 or 12/40/2015.
141      * Also enforces the year has to pass with 4 digits.
142      * @param value : the date and time
143      * @return : the correctly formatted date and time
144      * @throws ParseException : Exception while parsing the date and time
145      */
146     public static String checkDateTimeFormat(String value) throws ParseException {
147         if (value == null || !value.matches("^\\d{1,2}/\\d{1,2}/\\d{4} \\d{1,2}:\\d{2}$")) {
148             System.out.println("Please provide the date and time in format mm/dd/yyyy hh:mm");
149             System.exit(1);
150         }else{
151             SimpleDateFormat df = new SimpleDateFormat("MM/dd/yyyy HH:mm");
152             df.setLenient(false);
153            df.parse(value);
154         }
155         return value;
156     }
