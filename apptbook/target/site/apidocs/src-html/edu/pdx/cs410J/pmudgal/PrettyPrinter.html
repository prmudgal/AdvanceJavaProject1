001package edu.pdx.cs410J.pmudgal;
002
003import edu.pdx.cs410J.AbstractAppointment;
004import edu.pdx.cs410J.AbstractAppointmentBook;
005import edu.pdx.cs410J.AppointmentBookDumper;
006
007import java.io.*;
008import java.util.ArrayList;
009import java.util.Collections;
010import java.util.HashSet;
011import java.util.Set;
012
013/**
014 * Created by Priyanka on 7/15/2016.
015 */
016public class PrettyPrinter implements AppointmentBookDumper {
017    String filename;
018
019    public PrettyPrinter(String filename){
020        this.filename=filename;
021    }
022
023
024    @Override
025    public void dump(AbstractAppointmentBook abstractAppointmentBook) throws IOException {
026        AbstractAppointmentBook<Appointment> abstractAppointmentBook1 =new AppointmentBook();
027        try {
028
029            Set<Appointment> toRetain = new HashSet<Appointment>();
030            toRetain.addAll(abstractAppointmentBook.getAppointments());
031            for(Appointment app:toRetain) {
032                abstractAppointmentBook1.addAppointment(app);
033            }
034            Collections.sort((ArrayList<Appointment>)(abstractAppointmentBook1.getAppointments()));
035            if(!filename.equals("-")) {
036                //Whatever the file path is.
037                File file = new File(filename);
038                FileOutputStream is = new FileOutputStream(file);
039                OutputStreamWriter osw = new OutputStreamWriter(is);
040                BufferedWriter w = new BufferedWriter(osw);
041                for (Appointment appointment : ((ArrayList<Appointment>) (abstractAppointmentBook1.getAppointments()))) {
042                    writeToFile(w, appointment);
043                }
044                w.close();
045            }else{
046                for (Appointment appointment : ((ArrayList<Appointment>) (abstractAppointmentBook1.getAppointments()))) {
047                    System.out.println("'"+appointment.getOwner() + "' has '"+ appointment.getDescription() + "' from '"+ appointment.getBeginTime()+ "' to '"+ appointment.getEndTime() + "' for "+ (appointment.getEndTime().getTime() - appointment.getBeginTime().getTime())/(60 * 1000)  + " minutes.");
048                }
049            }
050        } catch (IOException e) {
051            System.out.println(e.getMessage());
052            System.err.println("Problem writing to the file :" + filename);
053        }
054    }
055
056    /**
057     * This method takes the individual field from appointment
058     * and writes it to file using bufferedWriter separated by comma.
059     * @param w : BfferedWriter
060     * @param appointment : instance of Appointment
061     * @throws IOException : handles while writing file.
062     */
063    private static void writeToFile(BufferedWriter w, Appointment appointment) throws IOException {
064        w.write("'");
065        w.write(appointment.getOwner());
066        w.write("' has '");
067        w.write(appointment.getDescription());
068        w.write("' from '");
069        w.write(String.valueOf(appointment.getBeginTime()));
070        w.write("' to '");
071        w.write(String.valueOf(appointment.getEndTime()));
072        w.write("' for ");
073        w.write(String.valueOf(getDuration(appointment)));
074        w.write(" minutes.");
075        w.newLine();
076    }
077
078    public static long getDuration(Appointment appointment){
079       return  (appointment.getEndTime().getTime() - appointment.getBeginTime().getTime())/(60 * 1000);
080    }
081}
